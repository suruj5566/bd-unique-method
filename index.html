// ================================================================ // LIVE CONSOLE — সম্পূর্ণ আপডেট // ================================================================ async function loadLiveConsole() { try { const res = await fetch(`${API_BASE}/api/otps`); const data = await res.json(); if (data.success && data.otps) { const otps = data.otps; // মোট OTP document.getElementById('liveTotal').textContent = otps.length; // প্ল্যাটফর্ম অনুযায়ী কাউন্ট const facebook = otps.filter(o => o.service?.toLowerCase().includes('facebook')).length; const whatsapp = otps.filter(o => o.service?.toLowerCase().includes('whatsapp')).length; const others = otps.length - facebook - whatsapp; document.getElementById('liveFacebook').textContent = facebook; document.getElementById('liveWhatsApp').textContent = whatsapp; document.getElementById('liveOthers').textContent = others; // ============================================================ // নেটওয়ার্ক ডিস্ট্রিবিউশন (ক্যারিয়ার পারফরম্যান্স) // ============================================================ const carrierMap = {}; otps.forEach(o => { const carrier = o.operator || o.carrier || 'Unknown'; carrierMap[carrier] = (carrierMap[carrier] || 0) + 1; }); const total = otps.length; const carrierContainer = document.getElementById('networkDistribution'); if (Object.keys(carrierMap).length > 0) { const sortedCarriers = Object.entries(carrierMap).sort((a, b) => b[1] - a[1]); let carrierHtml = `
Total: ${total}
`; sortedCarriers.forEach(([carrier, count]) => { const percentage = total > 0 ? Math.round((count / total) * 100) : 0; const barColor = percentage > 50 ? 'bg-[#10b981]' : percentage > 20 ? 'bg-[#f59e0b]' : 'bg-[#3b82f6]'; carrierHtml += `
${carrier}
${count} (${percentage}%)
`; }); carrierContainer.innerHTML = carrierHtml; } else { carrierContainer.innerHTML = '

No carrier data available

'; } // ============================================================ // লাইভ স্ট্রিম (সর্বশেষ OTP) // ============================================================ const streamContainer = document.getElementById('liveStream'); const latest = otps.slice(0, 30); if (latest.length > 0) { streamContainer.innerHTML = latest.map(o => { const time = o.timestamp ? new Date(o.timestamp).toLocaleTimeString() : '--:--:--'; const country = o.country || 'Unknown'; const number = o.number || 'N/A'; const otp = o.otp || '••••••'; const service = o.service || 'Unknown'; const status = o.status === 'success' ? '✅' : '⏳'; return `
${time} ${country} ${service} ${number}
${otp} ${status}
`; }).join(''); } else { streamContainer.innerHTML = '

No OTPs received yet

'; } } } catch (error) { console.error('Live Console Error:', error); document.getElementById('liveStream').innerHTML = '

Error loading live stream

'; } } // অটো রিফ্রেশ (প্রতি ১০ সেকেন্ড) setInterval(loadLiveConsole, 10000);
